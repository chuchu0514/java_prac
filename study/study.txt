 // 설계도 (클래스)
public class Dog {

    // === 필드 (변수들) ===
    private String name;
    private int age;
    private static int count = 0;  // static: 모든 Dog가 공유하는 변수

    // === 생성자 ===
    public Dog(String name, int age) {
        this.name = name;   // this.name = 필드, name = 매개변수
        this.age = age;
        count++;            // Dog 만들 때마다 카운트 증가
    }

    // 생성자는 여러 개 만들 수 있어 (오버로딩)
    public Dog(String name) {
        this.name = name;
        this.age = 0;       // age 없이 만들면 0으로
        count++;
    }

    // === 메서드 ===
    public void bark() {
        System.out.println(name + ": 멍!");
    }

    public String getName() { return name; }
    public int getAge() { return age; }

    public static int getCount() {   // static 메서드는 객체 없이 호출
        return count;
    }
}

다운캐스팅 — 업캐스팅된 걸 다시 자식 타입으로 되돌리는 것. 강제 변환 필요.
Person p = new Student("이재문");  // 업캐스팅
Student s = (Student) p;           // 다운캐스팅 — 괄호로 명시해야 함
s.grade = "A";  // ✅ 이제 Student 멤버 접근 가능

    abstract void draw();  // 구현 없음 — 자식이 반드시 구현해야 해

인터페이스
interface PhoneInterface {
    void sendCall();     // 이름만, 내용 없음
    void receiveCall();  // 이름만, 내용 없음
}

class SamsungPhone implements PhoneInterface {
    public void sendCall() { System.out.println("띠리리리링"); }    // 구현 안 하면 오류
    public void receiveCall() { System.out.println("전화 왔습니다."); } // 구현 안 하면 오류
}

class IPhone implements PhoneInterface {
    public void sendCall() { System.out.println("뚜루루루"); }      // 내용은 달라도 됨
    public void receiveCall() { System.out.println("call incoming"); }
}

interface PhoneInterface {
    int TIMEOUT = 10000;                  // 이거나
    final int TIMEOUT = 10000;            // 이거나
    public static final int TIMEOUT = 10000;  // 이거나
    // 셋 다 완전히 똑같아
}인터페이스 안에서는 어차피 자동으로 public static final이 붙거든.
